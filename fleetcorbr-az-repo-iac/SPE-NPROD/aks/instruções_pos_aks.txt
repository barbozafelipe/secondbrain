### INSTALAÇÃO NGINX INGRESS CONTROLER VIA Helm:

## Comandos básicos para criação:
helm repo add ingress-nginx https://kubernetes.github.io/ingress-nginx
helm repo update

## Instalação utilizando um IP Externo:

# Helm de instalação
helm install ingress-nginx ingress-nginx/ingress-nginx \
  --create-namespace \
  --namespace ingress-nginx \
  --set controller.service.annotations."service\.beta\.kubernetes\.io/azure-load-balancer-health-probe-request-path"=/healthz \
  --set controller.replicaCount=3 \
  --set controller.nodeSelector."kubernetes\.io/os"=linux \
  --set controller.admissionWebhooks.patch.nodeSelector."kubernetes\.io/os"=linux \
  --set controller.service.externalTrafficPolicy=Local \
  --set controller.service.loadBalancerIP=4.228.34.112 \
  --set controller.ingressClassResource.name=nginx \
  --set controller.ingressClass=nginx
  
 
# Compactado em 1 linha
helm install ingress-nginx ingress-nginx/ingress-nginx --create-namespace  --namespace ingress-nginx  --set controller.service.annotations."service\.beta\.kubernetes\.io/azure-load-balancer-health-probe-request-path"=/healthz --set controller.replicaCount=3 --set controller.nodeSelector."kubernetes\.io/os"=linux --set controller.admissionWebhooks.patch.nodeSelector."kubernetes\.io/os"=linux --set controller.service.externalTrafficPolicy=Local --set controller.service.loadBalancerIP=4.228.34.112 --set controller.ingressClass=nginx

# Usando o Echo-Server para testar
kubectl create -f https://raw.githubusercontent.com/kubernetes/ingress-nginx/main/docs/examples/http-svc.yaml

# Publicando um ingress
kubectl apply -f example-ingress.yaml

# Criação Secret com Certificado
kubectl create secret tls kafka-cert-v1-2027jun28 --key="tls.key" --cert="tls.crt" -n kube-system


## Instalação utilizando um IP Interno:

helm install ingress-nginx-internal ingress-nginx/ingress-nginx \
    --namespace ingress-nginx-internal \
    --create-namespace \
    --set controller.replicaCount=3 \
    --set controller.nodeSelector."kubernetes\.io/os"=linux \
    --set controller.admissionWebhooks.patch.nodeSelector."kubernetes\.io/os"=linux \
    --set controller.service.loadBalancerIP=10.17.203.254 \
    --set controller.service.annotations."service\.beta\.kubernetes\.io/azure-load-balancer-internal"=true \
    --set controller.service.annotations."service\.beta\.kubernetes\.io/azure-load-balancer-health-probe-request-path"=/healthz \
    --set defaultBackend.nodeSelector."kubernetes\.io/os"=linux \
    --set controller.service.externalTrafficPolicy=Local \
    --set controller.ingressClassResource.name=nginx-internal
    --set controller.ingressClass=nginx-internal
    --set controller.extraArgs.default-ssl-certificate="<namespace>/<secret_name>"
    --set "controller.extraArgs.enable-ssl-passthrough="

# Compactado em 1 linha
helm install ingress-nginx-internal ingress-nginx/ingress-nginx --namespace ingress-nginx-internal --create-namespace --set controller.replicaCount=3 --set controller.nodeSelector."kubernetes\.io/os"=linux --set controller.admissionWebhooks.patch.nodeSelector."kubernetes\.io/os"=linux --set controller.service.loadBalancerIP=10.17.203.254 --set controller.service.annotations."service\.beta\.kubernetes\.io/azure-load-balancer-internal"=true --set controller.service.annotations."service\.beta\.kubernetes\.io/azure-load-balancer-health-probe-request-path"=/healthz --set defaultBackend.nodeSelector."kubernetes\.io/os"=linux --set controller.service.externalTrafficPolicy=Local --set controller.ingressClassResource.name=nginx-internal --set controller.ingressClass=nginx-internal --set controller.extraArgs.default-ssl-certificate="kube-system/kafka-cert-v1-2027jun28"   --set "controller.extraArgs.enable-ssl-passthrough="

# Publicando um ingress
kubectl apply -f example-ingress-internal.yaml

### Instalação do Operator Strimzi Kafka

## Comando para Instalação:
kubectl create -f kafka/strimzi-kafka-operator.yaml
> O operator é instalado no namespace  "operators". (por padrão... alterar no yaml do Operator!)

## Validação:
kubectl get csv -n kafka

## Alguns comandos para verificação da Subscription:

# Listar a Subscription:
kubectl get sub -n kafka

# Listar o InstallPlan:
kubectl get ip -n kafka

# aplicar um patch para mudar o InstallPlan para Manual:
kubectl edit sub my-strimzi-kafka-operator -n operators
kubectl edit ip install-892rh -n operators -o yaml

## Para fazer atualização do Operator:
kubectl get ip -n operators
kubectl patch installplan install-2lxhr --type merge --patch '{"spec":{"approved":true}}'
kubectl get csv -n operators

## Para fazer o deploy do Cluster Kafka:


# Criação dos ConfigMaps:
kubectl apply -f kafka\configmap-kafka-metrics.yaml
kubectl apply -f kafka\configmap-zookeeper-metrics.yaml

# Criação Secret com Certificado
kubectl create secret tls kafka-cert-v1-2027jun28 --key="tls.key" --cert="tls.crt" -n kafka

# Instalação do Clsuter Kafka
kubectl apply -f kafka\kafka.yaml

# Análise dos Logs
kubectl logs -f -l name=strimzi-cluster-operator --all-containers -n kafka
